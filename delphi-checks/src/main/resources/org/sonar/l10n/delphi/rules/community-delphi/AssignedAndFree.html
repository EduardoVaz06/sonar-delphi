The TObject.Free method already checks for assignment before freeing. Additional checks are redundant. 

    
if Assigned(Foo) then begin // Noncompliant
  Foo.Free;
end;

if Foo <> nil then begin // Noncompliant
  Foo.Free;
end;

if Assigned(Foo) then begin // Noncompliant
  FreeAndNil(Foo);
end;

if Foo <> nil then begin // Noncompliant
  FreeAndNil(Foo);
end;

Foo.Free; // Compliant

FreeAndNil(Foo); // Compliant
