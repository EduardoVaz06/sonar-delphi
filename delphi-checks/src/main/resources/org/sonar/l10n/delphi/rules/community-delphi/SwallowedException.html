Exceptions should not be swallowed.
Exception handlers should contain logic to handle the caught exception. 

    
try
  RaiseException;
except	// Noncompliant
  // Do nothing
end;

try
  RaiseException;
except	// Noncompliant
  on E: Exception do begin
    // Do nothing
  end;
end;

try
  RaiseException;
except	// Compliant
  on E: Exception do begin
    Log.Audit('He''s dead, Jim!');
  end;
end;
