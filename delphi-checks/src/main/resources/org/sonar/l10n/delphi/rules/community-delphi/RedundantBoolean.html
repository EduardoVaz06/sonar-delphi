Redundant Boolean literals should be removed from expressions to improve readability.

    
if Foo = True then begin // Noncompliant
  DoSomething;
end;

if Bar = False then begin // Noncompliant
  DoSomething;
end;

DoSomething(not False); // Noncompliant

if Foo then begin // Compliant
  DoSomething;
end;

if not Bar then begin // Compliant
  DoSomething;
end;

DoSomething(True); // Compliant
